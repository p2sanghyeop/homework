select
d.device_uuid,
d.firmware_ver 
from devices d
inner join car_information ci 
on d.device_uuid = ci.device_uuid 
right join car_event_log cel 
on ci.car_number = cel.car_number 
where cel.event_type  >=date_format(cel.create_at,"2022-04-11")
